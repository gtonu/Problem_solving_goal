#include<bits/stdc++.h>
using namespace std;
int N=2e6+1;
vector<int> v(N,0);
void solve()
{
    int n;
    cin>>n;
    vector<int> v;   map<int,int> mp;
    for(int i=0;i<n;i++)
        {
            int x;
            cin>>x;
            v.push_back(x);
            mp[v[i]]++;
        }
        if(n<=2)
        {
            cout<<"-1"<<"\n";
            return;
        }
        int j;   bool flag=false;
        for(int i=1;i<=n;i++)
        {
            if(mp[i]>=3)
            {
                j=i;
                flag=true;
            }
        }
        if(flag)
         cout<<j<<"\n";
         else
            cout<<"-1"<<"\n";
}
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
       solve();
    }

    return 0;
}
